file_name <- read.csv("China_EO_49to17.csv")
china_data <- as_tibble(file_name)
china_data <- arrange(china_data, desc(Year))

shanghai_and_beijing <- china_data %>% 
                        filter(Year >= 2000) %>% 
                        select(Year,
                               Beijing_Enterprise,
                               Beijing_Output,
                               Shanghai_Enterprise,
                               Shanghai_Output)
